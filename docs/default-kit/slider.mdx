import { Slider } from '@react-three/uikit-default'

        overflow="scroll"
        flexDirection="column"
        alignItems="center"
        padding={32}
        backgroundColor={colors.background}
      >
        <SliderDemo />
      </Fullscreen>
    </Canvas>

}`}}
